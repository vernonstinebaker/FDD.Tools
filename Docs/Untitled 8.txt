    $('.my-button').on('click', function(e) {

        var shortcode = '[e2pdf-upload id="1" arg1="item-0"]'; // Replace with your desired shortcode

        $.ajax({
            url: my_ajax_object.ajax_url, // Defined by WordPress
            type: 'POST',
            data: {
                action: 'execute_shortcode',
                shortcode: shortcode
            },
            success: function(response) {
                console.log(response); // Handle the shortcode output as needed
            }
        });
    });
});
</script>
