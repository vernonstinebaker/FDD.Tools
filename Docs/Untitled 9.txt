SELECT 
  t1.user_id,
  t1.meta_key AS linkedin_meta_key,
  t1.meta_value AS linkedin_meta_value,
  t2.meta_key AS linkedin_profile_meta_key,
  t2.meta_value AS linkedin_profile_meta_value
FROM wp_usermeta t1
LEFT JOIN wp_usermeta t2
  ON t1.user_id = t2.user_id
  AND t2.meta_key = 'linkedin_profile'
WHERE t1.meta_key = 'linkedin'
  AND (t1.meta_value <> t2.meta_value OR t2.meta_value IS NULL)
ORDER BY t1.user_id;
