import SwiftUI

struct ParkingLotView: View {
    let features = [
        Feature(name: "Feature 1", status: .inProgress, completionPercentage: 0.5, dueDateString: "04/24"),
        Feature(name: "Feature 2", status: .inProgress, completionPercentage: 0.5, dueDateString:"06/24"),
        Feature(name: "Feature 3", status: .notStarted, completionPercentage: 0.0, dueDateString: "05/24"),
        Feature(name: "Feature 4", status: .atRisk, completionPercentage: 0.25, dueDateString: "06/24")
    ]
    
    var body: some View {
        VStack(alignment: .leading) {
            Text("Parking Lot Chart")
                .font(.largeTitle)
                .padding()
            
            ScrollView {
                LazyVGrid(columns: [GridItem(.adaptive(minimum: 160), spacing: 20)], spacing: 20) {
                    ForEach(features) { feature in
                        FeatureView(feature: feature)
                    }
                }
                .padding()
            }
        }
    }
}
